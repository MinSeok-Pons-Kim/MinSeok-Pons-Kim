import numpy as np 
def graph_solver(p):
    graph = []
    graph[0:50] = [nx.watts_strogatz_graph(n=500, k=4, p=p) for i in range(50)]
    d = [nx.average_shortest_path_length(i) for i in graph]
    ave_d = sum(d) / len(graph)
    ave_stds = np.std(d)
    print "average distance: ", ave_d
    print "average std: ", ave_stds
    print '\n'
    return ave_d, ave_stds

import numpy

def graph_drawer(graph):
    flatten_edges = np.array(graph.edges()).flatten()
    unique, counts = numpy.unique(flatten_edges, return_counts=True)
    proportion_list = counts/float(sum(counts))
    chosen_node = np.random.choice(graph.nodes(), p=proportion_list)
    graph.add_node(len(graph)+1)
    graph.add_edge(len(graph), chosen_node)
    return graph

graph = nx.Graph()
graph.add_nodes_from([1,2])
graph.add_edge(1,2)

for i in range(100):

Power-laws and the friendship paradox¶

Next step is to explore the Friendship paradox. This paradox states that almost everyone have fewer friends than their friends have, on average. This sounds crazy, but is actually an almost trivial consequence of living in a social network with a power-law degree distribution. The explanation is that almost everyone is friends with a hub, that drives up the average degree of the friends. Let's explore that in the 5000 node BA network we've just generated. Do the following:

  • Pick a node i at random (e.g. use random.choice). Find its degree.
  • Find i's neighbors. And calculate their average degree.
  • Compare the two numbers to check if it's true that i's friends (on average) have more friends than i.
  • Do this 1000 times. How many out of those 1000 times is the friendship paradox true?

import urllib
import re

def listWikiPhilos(link, deduction=0):
    # Crawl
    if "en.wikipedia.org" in link:
        # Link
        response = urllib.urlopen(link)
    else:
        # Keyword 
        response = urllib.urlopen("http://en.wikipedia.org/wiki/"+ link)
    wikiData = response.read()
    
    # Analyze with Regex
    if deduction:
        philosList = re.findall(r'<li><a href=\"/wiki/([^>]+)" [^>]+>([^>]+)</a', wikiData)[:-deduction]
    else:
        philosList = re.findall(r'<li><a href=\"/wiki/([^>]+)" [^>]+>([^>]+)</a', wikiData)
    return philosList

def philosCount(philosSet, philosCategories):
    philos_count = {}
    for philo in philosSet:
        philo_count = 0
        for philosCategory in philosCategories:
            if philo in philosCategory:
                philo_count += 1
                
        philos_count[philo] = philo_count
        
    return philos_count

class ModeError(Exception):
    pass


def freqPhilos(philosCount, mode='max', freq_count=2):
    if mode == 'max':
        max_count = max(philosCount.values())
    elif mode == 'freq':
        pass
    else:
        raise ModeError("Error message : Incorrect Mode")
        return -1
    freqPhilos = []
    for idx, philo in enumerate(philosCount):
        if mode == 'max':
            if philosCount[philo] == max_count:
                freqPhilos.append(philo)
        elif mode == 'freq':
            if philosCount[philo] >= freq_count:
                freqPhilos.append([philo, philosCount[philo]])
                                  
    if mode == 'max':
        return freqPhilos, max_count
    elif mode == 'freq':
        return freqPhilos
